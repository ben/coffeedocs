elemica.FilterHandler = 
{
  setupFilters: function() {
    return $('ul.filters li').on('click', 'button.status', function(event) {

      /*
      The attribute is undefined for the 'all' filter
       */
      var filters, _ref;
      filters = (_ref = $(event.target).attr('data-filter')) != null ? _ref.split(' ').slice(0) : void 0;
      $(event.target).parent().siblings('.current').removeClass('current').end().addClass('current');
      return elemica.event('filter-changed', {
        selectedFilters: filters
      });
    });
  },
  setupTagFilters: function() {
    return $('ul.tag-filters').on('click', 'button.tag', function(event) {
      var $tag, selectedTags;
      selectedTags = [];
      $tag = $(event.target).parent();
      $tag.toggleClass('current');
      if ($tag.is('.current')) {
        selectedTags.push(String($(event.target).data('filter')));
      }
      $tag.siblings('.current').find('button').each(function() {
        return selectedTags.push($(this).data('filter'));
      });
      return elemica.event('tag-filter-changed', {
        selectedTags: selectedTags
      });
    });
  }
}

Provides a setupFilters function that will set up the filters on the current page. Filters are set up to dispatch the filter-changed event when one is clicked, and that event carries a selectedFilters property with an array of filters that should be applied based on the data-filter attribute of the specific filter, split by space.
