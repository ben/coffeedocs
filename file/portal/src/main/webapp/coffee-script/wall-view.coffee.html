CommentHandler = 
{
  setupComments: function($wallElement) {
    $(document).on('comment-saved', function(event) {
      return $('.post-comment.submitting').each(function() {
        if (ko.dataFor(this).id() === event.parentEventId) {
          return $(this).trigger('form-submission-completed');
        }
      });
    });
    $wallElement.on('keydown', '.post-comment > textarea', function(event) {
      if (event.keyCode === ENTER && !(event.shiftKey || event.altKey)) {
        $(this).closest('.post-comment').submit();
        event.preventDefault();
        return event.stopImmediatePropagation();
      }
    });
    return elemica.autogrow($wallElement, '.post-comment > textarea');
  }
}
EntryActionHandler = 
{
  setupEntryActions: function($wallElement) {
    $(document).on('show-edit-message-modal', function(event) {
      $('#message-title').val(event.subject);
      $('#global-message-body').val(event.body);
      $('#notification-id').val(event.id);
      return $('#edit-global-messages').modal();
    }).on('cleanup-send-message', function(event) {
      $('#message-title, #message-body, #global-message-body, #notification-id').val('');
      $("#" + event.modalName).modal('hide');
      return false;
    });
    return $wallElement.on('click.collapse-comments.collapse-details', 'button[data-toggle^=collapse-]', function(event) {
      var $target, collapsingClass;
      collapsingClass = $(event.target).attr('data-toggle').replace(/^collapse-/, '');
      $target = $(this).closest('li.wall-entry').children("div." + collapsingClass);
      if ($target.data('collapse')) {
        return $target.collapse('toggle');
      } else {
        return $target.collapse();
      }
    }).on('click', 'button.edit', function() {
      var entry;
      entry = ko.dataFor(this);
      return entry.edit()();
    }).on('click', 'button.delete', function() {
      var entry;
      entry = ko.dataFor(this);
      if (confirm("Are you sure you want to do that?")) {
        return entry["delete"]()();
      }
    });
  }
}
